import { ChangeDetectorRef, Component, OnInit } from '@angular/core';
import { CommonModule } from '@angular/common';
import { FormsModule } from '@angular/forms';

import { TestPage } from '../../shared/test-page/test-page';
import { AssessmentService } from '../../../../core/services/assessment-service';
import { TestService } from '../../../../core/services/test-service';

// ================= Interfaces =================
export interface Topic {
  id: number;
  name: string;
  questionCount: number;
}

export interface Chapter {
  id: number;
  name: string;
  unlocked: boolean;
  completed: boolean;
  progress: number;
  topicsCount: number;
  questionsCount: number;
  topics: Topic[];
  isExpanded?: boolean;
}

export interface Unit {
  id: number;
  name: string;
  code: string;
  displayOrder: number;
  totalChapters: number;
  totalTopics: number;
  totalQuestions: number;
  progress: number;
  unlocked: boolean;
  completed: boolean;
  chapters: Chapter[];
}

export interface Subject {
  id: number;
  name: string;
  code: string;
  icon: string;
  color: string;
  units: Unit[];
}

@Component({
  selector: 'app-assessment',
  standalone: true,
  imports: [
    CommonModule,
    FormsModule,
    TestPage
  ],
  templateUrl: './assessment.html',
  styleUrl: './assessment.scss'
})
export class Assessment implements OnInit {

  constructor(
    private testService: TestService,
    private assessmentService: AssessmentService,
    private cdr: ChangeDetectorRef,
  ) { }

  // ================= Variables =================
  subjects: Subject[] = [];
  selectedSubject = '';
  currentChapters: Chapter[] = [];
  activeExamChapter: Chapter | null = null;
  isExamActive = false;

  // AI Modal Controller Variables
  isconformationpopup = false;
  pendingLaunchChapter: Chapter | null = null;

  // ================= Init =================
  ngOnInit(): void {
    console.log("is working ");
    this.loadAssessment();
  }

  // ================= API =================
  loadAssessment(): void {
    this.assessmentService.getAssessment().subscribe({
      next: (res: any) => {
        this.subjects = res.data || res;

        if (this.subjects.length > 0) {
          this.selectedSubject = this.subjects[0].name;
          this.updateCurrentChapters();
        }
        console.log(this.subjects);
        this.cdr.detectChanges();
      },
      error: err => {
        console.error(err);
      }
    });
  }

  // ================= Update Chapters Array Helper =================
  updateCurrentChapters(): void {
    const subject = this.subjects.find(s => s.name === this.selectedSubject);
    this.currentChapters = subject ? subject.units.flatMap(unit => unit.chapters) : [];
    this.cdr.detectChanges();
  }

  // ================= Subject Click =================
  selectSubject(name: string): void {
    if (this.isExamActive) {
      return;
    }
    this.selectedSubject = name;
    this.updateCurrentChapters();
    this.cdr.detectChanges();
  }

  // ================= Expand Chapter =================
  toggleChapter(chapter: Chapter): void {
    if (this.isExamActive) {
      return;
    }
    if (!chapter.unlocked) {
      return;
    }
    chapter.isExpanded = !chapter.isExpanded;
    this.cdr.detectChanges();
  }

  // ================= Launch Exam Interceptor =================
  launchExam(chapter: Chapter, event: Event): void {
    event.stopPropagation();
    if (!chapter.unlocked || this.isExamActive) {
      return;
    }

    // Capture chapter context and open confirmation UI
    this.pendingLaunchChapter = chapter;
    this.isconformationpopup = true;
    this.cdr.detectChanges();
  }

  // ================= Modal Decision Logic Actions =================
  // handleConfirm(): void {
  //   if (!this.pendingLaunchChapter) {
  //     this.handleCancel();
  //     return;
  //   }

  //   const targetChapter = this.pendingLaunchChapter;
  //   this.isconformationpopup = false;
  //   this.pendingLaunchChapter = null;

  //   // Execute backend sequence
  //   const testId: any = 1;
  //   this.testService.startTest(testId).subscribe({
  //     next: (res: any) => {
  //       if (res && res.success) {
  //         this.activeExamChapter = res;
  //         this.isExamActive = true;
  //         this.cdr.detectChanges();
  //       } else {
  //         console.error('Failed to start the exam:', res);
  //       }
  //     },
  //     error: (err: any) => {
  //       console.error('Error occurred while starting the exam:', err);
  //     }
  //   });
  // }


  // ================= Modal Decision Logic Actions =================
  handleConfirm(): void {
    if (!this.pendingLaunchChapter) {
      this.handleCancel();
      return;
    }

    // 1. Capture the true dynamic context object from state
    const targetChapter = this.pendingLaunchChapter;

    // Extract dynamic keys depending on your API structure (e.g., targetChapter.id)
    const chapterTestId = targetChapter.id;

    // 2. Reset the modal UI states immediately
    this.isconformationpopup = false;
    this.pendingLaunchChapter = null;

    // 3. Pass the dynamic chapter ID to your service call instead of a static number
    console.log("chapterTestId =-> "+chapterTestId);
      const body={
        testType:"CHAPTER_ASSESSMENT",
        referenceId:chapterTestId
      }

    this.testService.startTest(body).subscribe({
      next: (res: any) => {
        if (res && res.success) {
          this.activeExamChapter = res;
          this.isExamActive = true;
          this.cdr.detectChanges();
        } else {
          console.error('Failed to start the exam matching dynamic chapter:', res);
        }
      },
      error: (err: any) => {
        console.error('Error occurred while launching dynamic test sequence:', err);
      }
    });
  }
  handleCancel(): void {
    this.isconformationpopup = false;
    this.pendingLaunchChapter = null;
    this.cdr.detectChanges();
  }

  // ================= Finish Exam =================
  onExamFinished(scorePercentage: number): void {
    if (!this.activeExamChapter) {
      return;
    }

    if (scorePercentage >= 80) {
      this.subjects = this.subjects.map(subject => {
        const containsChapter = subject.units.some(unit =>
          unit.chapters.some(ch => ch.id === this.activeExamChapter!.id)
        );

        if (!containsChapter) return subject;

        return {
          ...subject,
          units: subject.units.map(unit => {
            const chaptersList = unit.chapters;
            const currentIndex = chaptersList.findIndex(ch => ch.id === this.activeExamChapter!.id);

            if (currentIndex === -1) return unit;

            const updatedChapters = chaptersList.map((ch, index) => {
              if (index === currentIndex) {
                return { ...ch, completed: true, progress: 100, isExpanded: ch.isExpanded };
              }
              if (index === currentIndex + 1) {
                return { ...ch, unlocked: true };
              }
              return ch;
            });

            return { ...unit, chapters: updatedChapters };
          })
        };
      });

      this.updateCurrentChapters();
    }

    this.activeExamChapter = null;
    this.isExamActive = false;
    this.loadAssessment();
    this.cdr.detectChanges();
  }

  // parent component
handleTestCancel(): void {
  this.isExamActive = false;

  // ================= Progress Color =================
  getProgressColor(progress: number): string {
    if (progress >= 100) {
      return '#22c55e';
    }
    if (progress >= 40) {
      return '#f59e0b';
    }
    return '#3b82f6';
  }
}
